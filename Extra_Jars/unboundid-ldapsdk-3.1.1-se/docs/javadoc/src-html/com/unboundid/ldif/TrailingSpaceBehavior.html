021    package com.unboundid.ldif;
022    
023    
024    
025    import com.unboundid.util.ThreadSafety;
026    import com.unboundid.util.ThreadSafetyLevel;
027    
028    
029    
030    /**
031     * This enum defines a set of possible behaviors that may be exhibited by the
032     * LDIF reader when encountering trailing spaces in attribute values that are
033     * not base64-encoded.
034     */
035    @ThreadSafety(level=ThreadSafetyLevel.COMPLETELY_THREADSAFE)
036    public enum TrailingSpaceBehavior
037    {
038      /**
039       * Indicates that illegal trailing spaces should be silently stripped from
040       * attribute values.
041       */
042      STRIP,
043    
044    
045    
046      /**
047       * Indicates that illegal trailing spaces should be retained (as if the value
048       * had been base64-encoded).
049       */
050      RETAIN,
051    
052    
053    
054      /**
055       * Indicates that illegal trailing spaces should cause the associated entry to
056       * be rejected.
057       */
058      REJECT;
059    }
